# IntroKotlin

fun main() {
    val fahrenheit = 350
    val celsius = (fahrenheit - 32) / 1.8
    println("La temperatura en grados Celsius es: $celsius")
}

fun main() {
    val peso = 3.0
    val meses = 6
    val dosisVacuna = ((peso + 10)/(meses * 10))*8
    println("La dosis de la vacuna que se debe aplicar es: $dosisVacuna")
}
